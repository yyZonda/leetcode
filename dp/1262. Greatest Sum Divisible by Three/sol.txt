class Solution {
    public int maxSumDivThree(int[] nums) {
        int sum = 0;
        int[] dp = new int[]{0,Integer.MIN_VALUE,Integer.MIN_VALUE};
        for(int i:nums){
            int[] dp2 = new int[3];
            for(int a=0;a<3;a++){
                dp2[(i+a)%3] = Math.max(dp[(i+a)%3],dp[a]+i);
            }
            dp=dp2;
        }
        return dp[0];
    }
}
